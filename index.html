// 初始位置設定在台中學校附近（台中火車站座標示例） const initialCoordinates = [24.1477, 120.6736]; let phoneId = localStorage.getItem("phoneId"); if (!phoneId) { phoneId = "phone-" + Math.random().toString(36).slice(2, 10); localStorage.setItem("phoneId", phoneId); } // 建立地圖並設置初始位置 let map = L.map('map').setView(initialCoordinates, 13); L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png').addTo(map); let marker = L.marker(initialCoordinates).addTo(map); // 定期更新位置 async function updateLocation() { try { let response = await fetch('http://你的VM外部IP:3000/api/locations'); let data = await response.json(); if (data && data.lat && data.lng) { // 如果獲得真實位置，更新地圖 map.setView([data.lat, data.lng], 13); marker.setLatLng([data.lat, data.lng]); } else { // 如果定位失敗，送 999，顯示固定位置 console.log("定位失敗，送 999"); map.setView(initialCoordinates, 13); marker.setLatLng(initialCoordinates); // 可以額外呼叫 API 送 999，例如： await fetch('http://你的VM外部IP:3000/api/locations', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ id: phoneId, lat: 999, lng: 999 }) }); } } catch (e) { console.error(e); console.log("發生錯誤，送 999"); map.setView(initialCoordinates, 13); marker.setLatLng(initialCoordinates); await fetch('http://你的VM外部IP:3000/api/locations', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ id: phoneId, lat: 999, lng: 999 }) }); } setTimeout(updateLocation, 5000); // 每5秒更新一次 } updateLocation();
